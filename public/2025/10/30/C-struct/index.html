1.从时间递增切入结构体

老实说，有点丑，而且不方便。如果日后功能拓展，我们想要递增的不是秒数，而是小时数，代码的框架结构就需要大改了

考虑以下做法

将时间递增逻辑抽象为函数，核心是按「秒 → 分 → 时 → 日 → 月 → 年」的优先级依次处理进位，同时需特殊处理月份天数差异和闰年 2 月的情况。

//z94-2

include<stdio.h>

typedef struct{
int year;
int month;
int day;
int hour;
int minute;
int second;
} Time;

int is_leap_year(int year){
return (year % 4 == 0 && year % 100 != 0) || (year % 400 == 0);
}

//Time *p = &t; // p 是结构体指针，指向变量 t（p 存储 t 的内存地址）
//time->member 等价于 (*time).member（先解引用指针得到结构体，再访问成员）。
int get_max_day(Time time){
switch (time->month) {
case 1: case 3: case 5: case 7: case 8: case 10: case 12:
return 31;
case 4: case 6: case 9: case 11:
return 30;
case 2:
return is_leap_year(time->year)? 29 : 28;
default:
return 0;
}
}
void incrementtime(Timetime) {
time->second++;
if(time->second == 60) {
time->second = 0;
time->minute++;

if(time->minute == 60) {
time->minute = 0;
time->hour++;

if(time->hour == 24){
time->hour = 0;
time->day++;

if(time->day > get_max_day(time)){
 time->day = 1;
 time->month++;

 if(time->month > 12){
  time->month = 1;
  time->year++;
 }
}

}
}
}
}

int main()
{
Time t;
scanf(“%d %d %d %d %d %d”, &t.year, &t.month, &t.day, &t.hour, &t.minute, &t.second);

incrementtime(&t);

printf("%d\n%d\n%d\n%d\n%d\n%d ", t.year, t.month, t.day, t.hour, t.minute, t.second);

return 0;
}
